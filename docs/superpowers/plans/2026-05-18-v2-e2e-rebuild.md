# V2 E2E Rebuild Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the default E2E signal with a deterministic V2 conversation PR gate while preserving and auditing the legacy tests.

**Architecture:** Add a V2-only E2E track under `tests/v2-e2e` with a controlled frontend test backend exposed only when `VITE_TEAMCLAW_E2E=true`. Tests drive real V2 React components and Zustand stores through `tauri-mcp` `executeJs`, while old `tests/e2e`, `tests/functional`, and `tests/regression` move behind an explicit legacy command and an audit matrix.

**Tech Stack:** Vitest, Tauri debug app, `tauri-plugin-mcp`, Zustand stores, generated TeamClaw protobuf messages, existing local-cache Tauri commands.

- [ ] **Step 1: Write the audit generator**

Create `scripts/audit-e2e-tests.mjs`:

```js
#!/usr/bin/env node
import { mkdirSync, readdirSync, readFileSync, writeFileSync } from "node:fs";
import { dirname, join } from "node:path";

const roots = ["tests/e2e", "tests/functional", "tests/regression"];
const outPath = "docs/testing/e2e-audit.md";

function walk(dir) {
  const entries = readdirSync(dir, { withFileTypes: true });
  return entries.flatMap((entry) => {
    const path = join(dir, entry.name);
    if (entry.isDirectory()) return walk(path);
    if (entry.isFile() && path.endsWith(".test.ts")) return [path];
    return [];
  });
}

function classify(file, content) {
  const lower = `${file}\n${content}`.toLowerCase();
  const assertions = (content.match(/expect\(/g) ?? []).length;
  const screenshots = (content.match(/takeScreenshot/g) ?? []).length;
  const hasOnlyWindowSmoke =
    assertions <= 2 &&
    /getWindowInfo/.test(content) &&
    /isVisible/.test(content) &&
    screenshots > 0;
  const hasCommentScript =
    /\/\/ 1\./.test(content) &&
    /\/\/ 2\./.test(content) &&
    assertions <= 2 &&
    screenshots > 0;

  if (hasCommentScript || hasOnlyWindowSmoke) {
    return ["delete", "Placeholder or screenshot/window-visible smoke without V2 behavioral assertion.", ""];
  }
  if (lower.includes("permission")) {
    return ["delete", "Permission Request live-thread rendering is disabled for current V2.", "Future Permission revival spec"];
  }
  if (
    lower.includes("install-script") ||
    lower.includes("auto update - configuration") ||
    lower.includes("release workflow") ||
    lower.includes("cargo.toml") ||
    lower.includes("tauri.conf.json")
  ) {
    return ["downgrade-unit", "Static configuration/source assertion; not an E2E user journey.", "tests/static or unit test"];
  }
  if (
    lower.includes("p2p") ||
    lower.includes("telemetry") ||
    lower.includes("sync") ||
    lower.includes("notification") ||
    lower.includes("gateway") ||
    lower.includes("channel")
  ) {
    return ["migrate-v2-nightly", "Environment-sensitive integration path; keep out of deterministic PR gate.", "V2 nightly real-chain or controlled integration"];
  }
  if (
    lower.includes("session") ||
    lower.includes("chat") ||
    lower.includes("stream") ||
    lower.includes("tool call") ||
    lower.includes("tool_call")
  ) {
    return ["migrate-v2-pr", "Conversation critical path overlaps the new V2 PR gate.", "tests/v2-e2e/pr/conversation-critical-path.test.ts"];
  }
  if (
    lower.includes("workspace") ||
    lower.includes("file") ||
    lower.includes("diff") ||
    lower.includes("markdown") ||
    lower.includes("terminal")
  ) {
    return ["keep-legacy-manual", "Useful non-conversation coverage, but outside phase-one V2 PR scope.", "Future V2 workspace/file track"];
  }
  if (lower.includes("settings") || lower.includes("knowledge") || lower.includes("layout")) {
    return ["keep-legacy-manual", "Useful product coverage, but outside phase-one V2 PR scope.", "Future V2 settings/layout track"];
  }
  return ["keep-legacy-manual", "Not part of the first V2 conversation gate; preserve manually until reviewed.", ""];
}

const rows = roots.flatMap((root) => walk(root)).sort().map((file) => {
  const content = readFileSync(file, "utf8");
  const currentTrack = file.split("/")[1];
  const [decision, reason, replacement] = classify(file, content);
  return { file, currentTrack, decision, reason, replacement };
});

const body = [
  "# E2E Audit Matrix",
  "",
  "Generated for the V2 E2E rebuild. Decisions are intentionally narrow so the default E2E signal can become trustworthy again.",
  "",
  "| File | Current Track | Decision | Reason | V2 Replacement | Notes |",
  "| --- | --- | --- | --- | --- | --- |",
  ...rows.map((row) =>
    `| \`${row.file}\` | \`${row.currentTrack}\` | \`${row.decision}\` | ${row.reason} | ${row.replacement ? `\`${row.replacement}\`` : ""} | Generated by \`scripts/audit-e2e-tests.mjs\`. |`
  ),
  "",
].join("\n");

mkdirSync(dirname(outPath), { recursive: true });
writeFileSync(outPath, body);
console.log(`Wrote ${outPath} with ${rows.length} rows`);
```

```ts
import { createMessage } from "@bufbuild/protobuf";
import {
  MessageKind,
  MessageSchema,
  type Message as TeamclawMessage,
} from "@/lib/proto/teamclaw_pb";
import {
  upsertActorsBatch,
  upsertSessionParticipantsBatch,
  type ActorRow,
  type SessionParticipantRow,
} from "@/lib/local-cache";
import { useSessionListStore, type SessionListEntry } from "@/stores/session-list-store";
import { useSessionStore } from "@/stores/session-store";
import { useUIStore } from "@/stores/ui";
import { useV2StreamingStore } from "@/stores/v2-streaming-store";
import { useWorkspaceStore } from "@/stores/workspace";

type ActorSeed = {
  id: string;
  teamId: string;
  actorType: "human" | "personal_agent" | "role_agent";
  displayName: string;
};

type SessionSeed = {
  id: string;
  title: string;
  teamId: string;
  lastMessageAt: string | null;
  lastMessagePreview: string | null;
  participantActorIds?: string[];
};

type MessageSeed = {
  id: string;
  sessionId: string;
  senderActorId: string;
  kind: "text" | "agent_reply" | "agent_tool_call" | "agent_tool_result" | "agent_thinking";
  content: string;
  createdAt: number;
  turnId?: string;
  model?: string;
  metadataJson?: string;
};

type ConversationSeed = {
  runId: string;
  teamId: string;
  workspacePath?: string;
  actors: ActorSeed[];
  sessions: SessionSeed[];
  messagesBySession?: Record<string, MessageSeed[]>;
  activeSessionId?: string;
};

function toEntry(seed: SessionSeed): SessionListEntry {
  return {
    id: seed.id,
    title: seed.title,
    team_id: seed.teamId,
    last_message_at: seed.lastMessageAt,
    last_message_preview: seed.lastMessagePreview,
    mode: "collab",
    idea_id: null,
    has_unread: false,
  };
}

function toKind(kind: MessageSeed["kind"]): MessageKind {
  switch (kind) {
    case "agent_reply":
      return MessageKind.AGENT_REPLY;
    case "agent_tool_call":
      return MessageKind.AGENT_TOOL_CALL;
    case "agent_tool_result":
      return MessageKind.AGENT_TOOL_RESULT;
    case "agent_thinking":
      return MessageKind.AGENT_THINKING;
    case "text":
    default:
      return MessageKind.TEXT;
  }
}

function toMessage(seed: MessageSeed): TeamclawMessage {
  return createMessage(MessageSchema, {
    messageId: seed.id,
    sessionId: seed.sessionId,
    senderActorId: seed.senderActorId,
    kind: toKind(seed.kind),
    content: seed.content,
    createdAt: BigInt(seed.createdAt),
    turnId: seed.turnId ?? "",
    model: seed.model ?? "",
    metadataJson: seed.metadataJson ?? "",
  });
}

function sortEntries(rows: SessionListEntry[]): SessionListEntry[] {
  return [...rows].sort((a, b) => {
    if (!a.last_message_at && !b.last_message_at) return 0;
    if (!a.last_message_at) return -1;
    if (!b.last_message_at) return 1;
    return b.last_message_at.localeCompare(a.last_message_at);
  });
}

async function seedConversation(seed: ConversationSeed) {
  const now = new Date().toISOString();
  const workspacePath = seed.workspacePath ?? `/tmp/teamclaw-v2-e2e-${seed.runId}`;

  useWorkspaceStore.setState({
    workspacePath,
    workspaceBootstrapped: true,
    workspaceReady: true,
  });

  const actorRows: ActorRow[] = seed.actors.map((actor) => ({
    id: actor.id,
    teamId: actor.teamId,
    actorType: actor.actorType,
    displayName: actor.displayName,
    avatarUrl: null,
    memberStatus: null,
    agentStatus: actor.actorType === "human" ? null : "active",
    metadataJson: null,
    createdAt: now,
    updatedAt: now,
    deletedAt: null,
    syncedAt: now,
  }));
  await upsertActorsBatch(actorRows);

  const participantRows: SessionParticipantRow[] = seed.sessions.flatMap((session) =>
    (session.participantActorIds ?? []).map((actorId) => ({
      id: `${session.id}:${actorId}`,
      sessionId: session.id,
      actorId,
      joinedAt: now,
      createdAt: now,
      updatedAt: now,
      deletedAt: null,
      syncedAt: now,
    })),
  );
  await upsertSessionParticipantsBatch(participantRows);

  useSessionListStore.setState({
    rows: sortEntries(seed.sessions.map(toEntry)),
    loading: false,
    error: null,
    hasMore: false,
    nextCursor: null,
  });

  const messages: Record<string, TeamclawMessage[]> = {};
  for (const [sessionId, seeds] of Object.entries(seed.messagesBySession ?? {})) {
    messages[sessionId] = seeds.map(toMessage);
  }

  useSessionStore.setState({
    messages,
    sessions: sortEntries(seed.sessions.map(toEntry)).map((entry) => ({
      ...entry,
      updatedAt: entry.last_message_at ? new Date(entry.last_message_at) : new Date(0),
      createdAt: entry.last_message_at ? new Date(entry.last_message_at) : new Date(0),
      messages: [],
      parentID: null,
      ideaId: entry.idea_id,
    })),
  });

  if (seed.activeSessionId) {
    useSessionStore.getState().setCurrent(seed.activeSessionId);
    useUIStore.setState({ activeSessionId: seed.activeSessionId });
  }

  return { ok: true, runId: seed.runId };
}

function switchSession(sessionId: string) {
  useSessionStore.getState().setCurrent(sessionId);
  useUIStore.setState({ activeSessionId: sessionId });
  return { ok: true, sessionId };
}

function appendMessage(seed: MessageSeed) {
  useSessionStore.getState().appendMessage(seed.sessionId, toMessage(seed));
  return { ok: true, messageId: seed.id };
}

function emitAgentDelta(args: { sessionId: string; actorId: string; delta: string }) {
  useV2StreamingStore.getState().appendOutput(args.sessionId, args.actorId, args.delta);
  return { ok: true };
}

function startTool(args: {
  sessionId: string;
  actorId: string;
  toolId: string;
  toolName: string;
  description: string;
  params?: Record<string, string>;
}) {
  useV2StreamingStore.getState().pushToolUse(args.sessionId, args.actorId, {
    toolId: args.toolId,
    toolName: args.toolName,
    description: args.description,
    params: args.params ?? {},
  });
  return { ok: true };
}

function completeTool(args: {
  sessionId: string;
  actorId: string;
  toolId: string;
  success: boolean;
  summary: string;
}) {
  useV2StreamingStore.getState().completeToolUse(args.sessionId, args.actorId, args);
  return { ok: true };
}

function setAgentError(args: { sessionId: string; actorId: string; message: string; details: string }) {
  useV2StreamingStore.getState().setError(args.sessionId, args.actorId, args.message, args.details);
  return { ok: true };
}

function completeRun(args: {
  sessionId: string;
  actorId: string;
  messageId: string;
  content: string;
  createdAt: number;
  turnId: string;
  model?: string;
}) {
  useV2StreamingStore.getState().finalize(args.sessionId, args.actorId, args.content);
  appendMessage({
    id: args.messageId,
    sessionId: args.sessionId,
    senderActorId: args.actorId,
    kind: "agent_reply",
    content: args.content,
    createdAt: args.createdAt,
    turnId: args.turnId,
    model: args.model ?? "e2e-model",
  });
  useV2StreamingStore.getState().clearActor(args.sessionId, args.actorId);
  useSessionListStore.setState((state) => ({
    rows: sortEntries(state.rows.map((row) =>
      row.id === args.sessionId
        ? {
            ...row,
            last_message_at: new Date(args.createdAt * 1000).toISOString(),
            last_message_preview: args.content,
          }
        : row,
    )),
  }));
  return { ok: true };
}

function cleanup() {
  useSessionListStore.setState({ rows: [], loading: false, error: null, hasMore: false, nextCursor: null });
  useSessionStore.setState({
    messages: {},
    sessions: [],
    currentSessionId: null,
    activeSessionId: null,
    pinnedSessionIds: [],
    highlightedSessionIds: [],
  });
  useV2StreamingStore.setState({ byKey: {} });
  useUIStore.setState({ activeSessionId: null });
  return { ok: true };
}

export function installV2E2EControl() {
  if (import.meta.env.VITE_TEAMCLAW_E2E !== "true") return;
  if (typeof window === "undefined") return;
  window.__TEAMCLAW_V2_E2E__ = {
    seedConversation,
    switchSession,
    appendMessage,
    emitAgentDelta,
    startTool,
    completeTool,
    setAgentError,
    completeRun,
    cleanup,
  };
}

declare global {
  interface Window {
    __TEAMCLAW_V2_E2E__?: {
      seedConversation: typeof seedConversation;
      switchSession: typeof switchSession;
      appendMessage: typeof appendMessage;
      emitAgentDelta: typeof emitAgentDelta;
      startTool: typeof startTool;
      completeTool: typeof completeTool;
      setAgentError: typeof setAgentError;
      completeRun: typeof completeRun;
      cleanup: typeof cleanup;
    };
  }
}
```

```ts
export async function v2Call<T = unknown>(method: string, args?: unknown): Promise<T> {
  const source = `
    (async () => {
      const api = window.__TEAMCLAW_V2_E2E__;
      if (!api) throw new Error("V2 E2E control surface is not installed. Build with VITE_TEAMCLAW_E2E=true.");
      const result = await api[${JSON.stringify(method)}](${JSON.stringify(args ?? null)});
      return JSON.stringify(result ?? null);
    })()
  `;
  const raw = await executeJs(source);
  return JSON.parse(raw) as T;
}

export async function cleanupV2() {
  try {
    await v2Call("cleanup");
  } catch {
    // App may not have launched far enough to install the control surface.
  }
}

export async function domText(selector: string): Promise<string> {
  const raw = await executeJs(`document.querySelector(${JSON.stringify(selector)})?.textContent ?? ""`);
  return raw;
}

export async function domCount(selector: string): Promise<number> {
  const raw = await executeJs(`String(document.querySelectorAll(${JSON.stringify(selector)}).length)`);
  return Number(raw);
}

export async function waitForText(text: string, timeoutMs = 10_000) {
  await waitForCondition(

    (value) => value === "true",
    timeoutMs,
    250,
  );
}

  await waitForCondition(
    `document.querySelector(${JSON.stringify(selector)}) ? "true" : "false"`,
    (value) => value === "true",
    timeoutMs,
    250,
  );
}

    })()
  `);
  expect(result).toBe("clicked");
}

    })()
  `);
  expect(result).toBe("set");
}

export async function submitComposer() {
  const result = await executeJs(`

      if (!button) return "missing";
      button.dispatchEvent(new MouseEvent("click", { bubbles: true, cancelable: true }));
      return "clicked";
    })()
  `);
  expect(result).toBe("clicked");
}
```
